#### 2.2 Council Execution Engine

```typescript
// src/shared/lib/council/engine.ts

export class CouncilEngine {
  constructor(private gateway: LLMGateway) {}
  
  async execute(
    council: Council, 
    prompt: string,
    onProgress?: (stage: number, progress: number) => void
  ): Promise<CouncilResult> {
    
    // Stage 1: Divergence (parallel)
    onProgress?.(1, 0);
    const stage1 = await this.executeDivergence(council, prompt);
    onProgress?.(1, 100);
    
    // Stage 2: Convergence (parallel evaluations)
    onProgress?.(2, 0);
    const stage2 = await this.executeConvergence(council, stage1);
    onProgress?.(2, 100);
    
    // Stage 3: Synthesis (chairman)
    onProgress?.(3, 0);
    const stage3 = await this.executeSynthesis(council, stage1, stage2, prompt);
    onProgress?.(3, 100);
    
    return { stage1, stage2, stage3, ... };
  }
  
  private async executeDivergence(council: Council, prompt: string) {
    // Fan-out: query all members in parallel
    const responses = await Promise.all(
      council.members.map(model => 
        this.gateway.query({ model: model.id, messages: [{ role: 'user', content: prompt }] })
      )
    );
    return { responses, totalLatencyMs: Math.max(...responses.map(r => r.latencyMs)) };
  }
  
  private async executeConvergence(council: Council, stage1: Stage1Result) {
    // Anonymize responses
    const anonymized = stage1.responses.map((r, i) => ({
      label: `Response ${String.fromCharCode(65 + i)}`, // A, B, C...
      content: r.content
    }));
    
    // Each evaluator ranks all responses
    const evaluations = await Promise.all(
      council.evaluators.map(evaluator =>
        this.evaluateResponses(evaluator, anonymized)
      )
    );
    
    // Aggregate rankings
    const aggregatedRanking = this.aggregateRankings(evaluations);
    const agreementScore = this.calculateAgreement(evaluations);
    
    return { evaluations, aggregatedRanking, agreementScore };
  }
  
  private async executeSynthesis(
    council: Council, 
    stage1: Stage1Result, 
    stage2: Stage2Result,
    originalPrompt: string
  ) {
    const synthesisPrompt = this.buildSynthesisPrompt(
      originalPrompt, 
      stage1, 
      stage2
    );
    
    const response = await this.gateway.query({
      model: council.chairman.id,
      messages: [{ role: 'user', content: synthesisPrompt }]
    });
```
